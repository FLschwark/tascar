Optional additional OSC server, e.g., for simultaneous access via TCP
and UDP.

\begin{tscattributes}
  \indattr{srv\_addr} & OSC server address for multicasting (or empty for unicast)\\
  \indattr{srv\_port} & OSC server port number (default: 9877)\\
  \indattr{srv\_proto} & OSC transport protocol, ``UDP'' or ``TCP'' (default: ``TCP'')\\
\end{tscattributes}
